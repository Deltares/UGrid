include(${CMAKE_SOURCE_DIR}/cmake/install_tools.cmake)

set(TOOLS_INSTALL_DIR ${CMAKE_CURRENT_BINARY_DIR}/tools)

#install swig
set(SWIG_VERSION 4.2.1)
set(SWIG_INSTALL_PREFIX ${TOOLS_INSTALL_DIR}/swig)
install_swig(
  SWIG_VERSION ${SWIG_VERSION}
  SWIG_INSTALL_PREFIX ${SWIG_INSTALL_PREFIX}
  SWIG_CLEAN FALSE
  SWIG_DIR_OUT SWIG_DIR
  SWIG_EXECUTABLE_OUT SWIG_EXECUTABLE
)
set(SWIG_DIR "${SWIG_DIR}" PARENT_SCOPE)
set(SWIG_EXECUTABLE "${SWIG_EXECUTABLE}" PARENT_SCOPE)

#install nuget
set(NUGET_VERSION "latest")
set(NUGET_INSTALL_PREFIX ${TOOLS_INSTALL_DIR})
install_nuget(
  NUGET_VERSION ${NUGET_VERSION}
  NUGET_INSTALL_PREFIX ${NUGET_INSTALL_PREFIX}
  NUGET_EXECUTABLE_OUT NUGET_EXECUTABLE
)
set(NUGET_EXECUTABLE "${NUGET_EXECUTABLE}" PARENT_SCOPE)
