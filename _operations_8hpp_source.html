107  static bool fill_ugrid_entity_dimension(std::multimap<std::string, netCDF::NcDim> const& dimensions,
108  std::string const& attribute_key_string,
109  std::string const& attribute_value_string,
110  std::map<UGridFileDimensions, netCDF::NcDim>& entity_dimensions)
111  {
112  auto const substring_dimension_pos = attribute_value_string.find("_dimension");
113  bool isDimensionVariable = false;
114  if (attribute_value_string.find("_dimension") != std::string::npos)
115  {
116  isDimensionVariable = true;
117  std::string location_name = attribute_value_string.substr(0, substring_dimension_pos);
118  auto const it = dimensions.find(attribute_key_string);
119  auto const dimension_enum = from_location_string_to_dimension(location_name);
120  entity_dimensions.insert({dimension_enum, it->second});
121  }
122  return isDimensionVariable;
123  }

125  static std::tuple<std::map<std::string, std::vector<netCDF::NcVar>>,
126  std::map<std::string, std::vector<std::string>>,
127  std::map<UGridFileDimensions, netCDF::NcDim>>
128  get_ugrid_entity(
129  netCDF::NcVar const& variable,
130  std::multimap<std::string, netCDF::NcDim> const& file_dimensions,
131  std::multimap<std::string, netCDF::NcVar> const& file_variables)
132  {
133  std::map<std::string, std::vector<netCDF::NcVar>> entity_attribute_variables;
134  std::map<std::string, std::vector<std::string>> entity_attribute_names;
135  std::map<UGridFileDimensions, netCDF::NcDim> entity_dimensions;
136  const auto variable_attributes = variable.getAtts();
137  for (const auto& attribute : variable_attributes)
138  {
139  if (attribute.second.getType() != netCDF::NcType::nc_CHAR)
140  {
141  continue;
142  }
143 
144  std::string attribute_key_string = attribute.first;
145  std::string attribute_value_string;
146  attribute.second.getValues(attribute_value_string);
147 
148  // check if it is a dimension variable
149  auto const isDimensionVariable = fill_ugrid_entity_dimension(file_dimensions, attribute_value_string, attribute_key_string, entity_dimensions);
150  if (isDimensionVariable)
151  {
152  continue;
153  }
154 
155  std::vector<std::string> attribute_value_string_tokens;
156  split(attribute_value_string_tokens, attribute_value_string, boost::is_any_of(" "));
157  std::vector<std::string> valid_variable_names;
158  std::vector<netCDF::NcVar> valid_attribute_variables;
159  for (auto const& token : attribute_value_string_tokens)
160  {
161  // a name of a valid attribute has been found
162  const auto it = file_variables.find(token);
163  if (it != file_variables.end())
164  {
165  valid_attribute_variables.emplace_back(it->second);
166  }
167  }
168  // valid file_variables have been found
169  if (!valid_attribute_variables.empty())
170  {
171  entity_attribute_variables.insert({attribute_key_string, valid_attribute_variables});
172  }
173 
174  entity_attribute_names.insert({attribute_key_string, attribute_value_string_tokens});
175  }
176  return {entity_attribute_variables, entity_attribute_names, entity_dimensions};
177  }

179  static void string_to_char_array(char* char_array, std::string const& value, size_t len)
180  {
181  if (char_array == nullptr || value.empty())
182  {
183  return;
184  }
185  for (auto i = 0; i < value.size(); ++i)
186  {
187  char_array[i] = value[i];
188  }
189  for (auto i = value.size(); i < len - 1; ++i)
190  {
191  char_array[i] = ' ';
192  }
193  char_array[len - 1] = '\0';
194  }

197  {
198  s.erase(std::find_if(s.rbegin(), s.rend(), [](auto& ch) {

205  {
