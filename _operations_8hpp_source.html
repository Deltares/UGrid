120 static std::tuple<std::map<std::string, std::vector<netCDF::NcVar>>,
121 std::map<std::string, std::vector<std::string>>,
122 std::map<UGridFileDimensions, netCDF::NcDim>>
123 get_ugrid_entity(netCDF::NcVar const& variable,
124 std::multimap<std::string, netCDF::NcDim> const& file_dimensions,
125 std::multimap<std::string, netCDF::NcVar> const& file_variables)
126 {
127 std::map<std::string, std::vector<netCDF::NcVar>> entity_attribute_variables;
128 std::map<std::string, std::vector<std::string>> entity_attribute_names;
129 std::map<UGridFileDimensions, netCDF::NcDim> entity_dimensions;
130 const auto variable_attributes = variable.getAtts();
131 for (const auto& attribute : variable_attributes)
132 {
133 if (attribute.second.getType() != netCDF::NcType::nc_CHAR)
134 {
135 continue;
136 }
137
138 std::string attribute_key_string = attribute.first;
139 std::string attribute_value_string;
140 attribute.second.getValues(attribute_value_string);
141
142 // check if it is a dimension variable
143 if (is_dimension_variable(file_dimensions, attribute_value_string))
144 {
145 continue;
146 }
147
148 std::vector<std::string> attribute_value_string_tokens;
149 split(attribute_value_string_tokens, attribute_value_string, boost::is_any_of(" "));
150 std::vector<std::string> valid_variable_names;
151 std::vector<netCDF::NcVar> valid_attribute_variables;
152 for (auto const& token : attribute_value_string_tokens)
153 {
154 // a name of a valid attribute has been found
155 const auto it = file_variables.find(token);
156 if (it != file_variables.end())
157 {
158 valid_attribute_variables.emplace_back(it->second);
159 }
160 }
161 // valid file_variables have been found
162 if (!valid_attribute_variables.empty())
163 {
164 entity_attribute_variables.insert({attribute_key_string, valid_attribute_variables});
165 }
166 entity_attribute_names.insert({attribute_key_string, attribute_value_string_tokens});
167 }
168
169 // get the dimensions from the variable dimensions
170 auto variable_value = entity_attribute_variables.find("node_coordinates");
171 if (variable_value != entity_attribute_variables.end())
172 {
173 entity_dimensions[UGridFileDimensions::node] = variable_value->second[0].getDims()[0];
174 }
175 variable_value = entity_attribute_variables.find("edge_node_connectivity");
176 if (variable_value != entity_attribute_variables.end())
177 {
178 entity_dimensions[UGridFileDimensions::edge] = variable_value->second[0].getDims()[0];
179 }
180 variable_value = entity_attribute_variables.find("face_coordinates");
181 if (variable_value != entity_attribute_variables.end())
182 {
183 entity_dimensions[UGridFileDimensions::face] = variable_value->second[0].getDims()[0];
184 }
185 variable_value = entity_attribute_variables.find("face_node_connectivity");
186 if (variable_value != entity_attribute_variables.end())
187 {
188 entity_dimensions[UGridFileDimensions::max_face_node] = variable_value->second[0].getDims()[1];
189 }
190
191 return {entity_attribute_variables, entity_attribute_names, entity_dimensions};
192 }

194 static void string_to_char_array(std::string const& value, size_t len, char* char_array)
195 {
196 if (char_array == nullptr || value.empty())
197 {
198 return;
199 }
200 for (size_t i = 0; i < value.size(); ++i)
201 {
202 char_array[i] = value[i];
203 }
204 for (auto i = value.size(); i < len - 1; ++i)
205 {
206 char_array[i] = ' ';
207 }
208
209 // null terminate
210 char_array[len - 1] = '\0';
211 }

213 static void vector_of_strings_to_char_array(std::vector<std::string> const& values, size_t len, char* char_array)
214 {
215 if (char_array == nullptr || values.empty())
216 {
217 return;
218 }
219
220 size_t char_array_position = 0;
221 for (size_t i = 0; i < values.size(); ++i)
222 {
223 for (size_t j = 0; j < values[i].size(); ++j)
224 {
225 char_array[char_array_position] = values[i][j];
226 char_array_position++;
227 }
228 for (size_t j = values[i].size(); j < len; ++j)
229 {
230 char_array[char_array_position] = ' ';
231 char_array_position++;
232 }
233 }
234
235 // null terminate
236 char_array[char_array_position - 1] = '\0';
237 }

239 static void right_trim_string(std::string& str)
240 {
241 auto const isalnum_lambda = [](auto const& ch)
242 {
243 return std::isalnum(ch);
244 };
245 str.erase(std::find_if(str.rbegin(), str.rend(), isalnum_lambda).base(), str.end());
246 }

248 static std::string char_array_to_string(char const* const char_array, size_t len)
249 {
250 std::string result(len, ' ');
251 if (char_array == nullptr)
252 {
253 return result;
254 }
255 for (size_t i = 0; i < len; ++i)
256 {
257 result[i] = char_array[i];
258 }
259 right_trim_string(result);
260 return result;
261 }
