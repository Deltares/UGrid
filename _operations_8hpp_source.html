125  static std::tuple<std::map<std::string, std::vector<netCDF::NcVar>>,
126  std::map<std::string, std::vector<std::string>>,
127  std::map<UGridFileDimensions, netCDF::NcDim>>
128  get_ugrid_entity(netCDF::NcVar const& variable,
129  std::multimap<std::string, netCDF::NcDim> const& file_dimensions,
130  std::multimap<std::string, netCDF::NcVar> const& file_variables)
131  {
132  std::map<std::string, std::vector<netCDF::NcVar>> entity_attribute_variables;
133  std::map<std::string, std::vector<std::string>> entity_attribute_names;
134  std::map<UGridFileDimensions, netCDF::NcDim> entity_dimensions;
135  const auto variable_attributes = variable.getAtts();
136  for (const auto& attribute : variable_attributes)
137  {
138  if (attribute.second.getType() != netCDF::NcType::nc_CHAR)
139  {
140  continue;
141  }
142 
143  std::string attribute_key_string = attribute.first;
144  std::string attribute_value_string;
145  attribute.second.getValues(attribute_value_string);
146 
147  // check if it is a dimension variable
148  if (is_dimension_variable(file_dimensions, attribute_value_string))
149  {
150  continue;
151  }
152 
153  std::vector<std::string> attribute_value_string_tokens;
154  split(attribute_value_string_tokens, attribute_value_string, boost::is_any_of(" "));
155  std::vector<std::string> valid_variable_names;
156  std::vector<netCDF::NcVar> valid_attribute_variables;
157  for (auto const& token : attribute_value_string_tokens)
158  {
159  // a name of a valid attribute has been found
160  const auto it = file_variables.find(token);
161  if (it != file_variables.end())
162  {
163  valid_attribute_variables.emplace_back(it->second);
164  }
165  }
166  // valid file_variables have been found
167  if (!valid_attribute_variables.empty())
168  {
169  entity_attribute_variables.insert({attribute_key_string, valid_attribute_variables});
170  }
171  entity_attribute_names.insert({attribute_key_string, attribute_value_string_tokens});
172  }
173 
174  // get the dimensions from the variable dimensions
175  auto variable_value = entity_attribute_variables.find("node_coordinates");
176  if (variable_value != entity_attribute_variables.end())
177  {
178  entity_dimensions[UGridFileDimensions::node] = variable_value->second[0].getDims()[0];
179  }
180  variable_value = entity_attribute_variables.find("edge_node_connectivity");
181  if (variable_value != entity_attribute_variables.end())
182  {
183  entity_dimensions[UGridFileDimensions::edge] = variable_value->second[0].getDims()[0];
184  }
185  variable_value = entity_attribute_variables.find("face_coordinates");
186  if (variable_value != entity_attribute_variables.end())
187  {
188  entity_dimensions[UGridFileDimensions::face] = variable_value->second[0].getDims()[0];
189  }
190  variable_value = entity_attribute_variables.find("face_node_connectivity");
191  if (variable_value != entity_attribute_variables.end())
192  {
193  entity_dimensions[UGridFileDimensions::max_face_node] = variable_value->second[0].getDims()[1];
194  }
195 
196  return {entity_attribute_variables, entity_attribute_names, entity_dimensions};
197  }

199  static void string_to_char_array(std::string const& value, size_t len, char* char_array)
200  {
201  if (char_array == nullptr || value.empty())
202  {
203  return;
204  }
205  for (auto i = 0; i < value.size(); ++i)
206  {
207  char_array[i] = value[i];
208  }
209  for (auto i = value.size(); i < len - 1; ++i)
210  {
211  char_array[i] = ' ';
212  }
213 
214  // null terminate
215  char_array[len - 1] = '\0';
216  }

218  static void vector_of_strings_to_char_array(std::vector<std::string> const& values, size_t len, char* char_array)
219  {
220  if (char_array == nullptr || values.empty())
221  {
222  return;
223  }
224 
225  size_t char_array_position = 0;
226  for (auto i = 0; i < values.size(); ++i)
227  {
228  for (auto j = 0; j < values[i].size(); ++j)
229  {
230  char_array[char_array_position] = values[i][j];
231  char_array_position++;
232  }
233  for (auto j = values[i].size(); j < len; ++j)
234  {
235  char_array[char_array_position] = ' ';
236  char_array_position++;
237  }
238  }
239 
240  // null terminate
241  char_array[char_array_position - 1] = '\0';
242  }

244  static void right_trim_string(std::string& str)
245  {
246  auto const isalnum_lambda = [](auto const& ch) { return std::isalnum(ch); };
247  str.erase(std::find_if(str.rbegin(), str.rend(), isalnum_lambda).base(), str.end());
