71  template <typename T>
72  [[nodiscard]] static std::vector<T> create(std::shared_ptr<netCDF::NcFile> const& nc_file)
73  {
74  // gets all variables in this file
75  auto const file_variables = nc_file->getVars();
76  auto const file_dimensions = nc_file->getDims();
77 
78  std::vector<T> result;
79  for (auto const& variable : file_variables)
80  {
81  auto variable_attributes = variable.second.getAtts();
82 
83  if (!T::is_topology_variable(variable_attributes))
84  {
85  continue;
86  }
87 
88  if (!T::has_matching_dimensionality(variable_attributes, T::get_dimensionality()))
89  {
90  continue;
91  }
92 
93  auto const [entity_attribute_variables, entity_attribute_names, entity_dimensions] = get_ugrid_entity(variable.second, file_dimensions, file_variables);
94  result.emplace_back(nc_file,
95  variable.second,
96  entity_attribute_variables,
97  entity_attribute_names,
98  entity_dimensions);
99  }
100 
101  return result;
102  }
