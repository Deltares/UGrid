65  template <typename T>
66  [[nodiscard]] static std::vector<T> create(std::shared_ptr<netCDF::NcFile> const& nc_file)
67  {
68  // gets all variables in this file
69  auto const file_variables = nc_file->getVars();
70  auto const file_dimensions = nc_file->getDims();
71 
72  std::vector<T> result;
73  for (auto const& variable : file_variables)
74  {
75  auto variable_attributes = variable.second.getAtts();
76 
77  if (!T::is_topology_variable(variable_attributes))
78  {
79  continue;
80  }
81 
82  if (!T::has_matching_dimensionality(variable_attributes, T::get_dimensionality()))
83  {
84  continue;
85  }
86 
87  auto const [entity_attribute_variables, entity_attribute_names, entity_dimensions] = get_ugrid_entity(variable.second, file_dimensions, file_variables);
88  result.emplace_back(nc_file,
89  variable.second,
90  entity_attribute_variables,
91  entity_attribute_names,
92  entity_dimensions);
93  }
94 
95  return result;
96  }
