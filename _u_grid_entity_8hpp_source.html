69 template <typename T>
70 [[nodiscard]] static std::vector<T> create(std::shared_ptr<netCDF::NcFile> const& nc_file)
71 {
72 // gets all variables in this file
73 auto const file_variables = nc_file->getVars();
74 auto const file_dimensions = nc_file->getDims();
75
76 std::vector<T> result;
77 for (auto const& variable : file_variables)
78 {
79 auto variable_attributes = variable.second.getAtts();
80
81 if (!T::is_topology_variable(variable_attributes))
82 {
83 continue;
84 }
85
86 if (!T::has_matching_dimensionality(variable_attributes, T::get_dimensionality()))
87 {
88 continue;
89 }
90
91 auto const [entity_attribute_variables, entity_attribute_names, entity_dimensions] = get_ugrid_entity(variable.second, file_dimensions, file_variables);
92 result.emplace_back(nc_file,
93 variable.second,
94 entity_attribute_variables,
95 entity_attribute_names,
96 entity_dimensions);
97 }
98
99 return result;
100 }

151 template <typename T>
152 static void apply_start_index_offset(const netCDF::NcVar& var, int start_index, int values_size, T* values)
153 {
154 const auto varAtt = var.getAtts();
155 std::string start_index_att_name{"start_index"};
156 if (varAtt.find(start_index_att_name) != varAtt.end())
157 {
158 int variable_start_index;
159 varAtt.at(start_index_att_name).getValues(&variable_start_index);
160 const int offset = start_index - variable_start_index;
161 for (int i = 0; i < values_size; ++i)
162 {
163 values[i] += offset;
164 }
165 }
166 }
