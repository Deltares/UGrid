67  template <typename T>
68  [[nodiscard]] static std::vector<T> create(std::shared_ptr<netCDF::NcFile> const& nc_file)
69  {
70  // gets all variables in this file
71  auto const file_variables = nc_file->getVars();
72  auto const file_dimensions = nc_file->getDims();
73 
74  std::vector<T> result;
75  for (auto const& variable : file_variables)
76  {
77  auto variable_attributes = variable.second.getAtts();
78 
79  if (!T::is_topology_variable(variable_attributes))
80  {
81  continue;
82  }
83 
84  if (!T::has_matching_dimensionality(variable_attributes, T::get_dimensionality()))
85  {
86  continue;
87  }
88 
89  auto const [entity_attribute_variables, entity_attribute_names, entity_dimensions] = get_ugrid_entity(variable.second, file_dimensions, file_variables);
90  result.emplace_back(nc_file,
91  variable.second,
92  entity_attribute_variables,
93  entity_attribute_names,
94  entity_dimensions);
95  }
96 
97  return result;
98  }

149  template <typename T>
150  static void apply_start_index_offset(const netCDF::NcVar& var, int start_index, int values_size, T* values)
151  {
152  const auto varAtt = var.getAtts();
153  std::string start_index_att_name{"start_index"};
154  if (varAtt.find(start_index_att_name) != varAtt.end())
155  {
156  int variable_start_index;
157  varAtt.at(start_index_att_name).getValues(&variable_start_index);
158  const int offset = start_index - variable_start_index;
159  for (int i = 0; i < values_size; ++i)
160  {
161  values[i] += offset;
162  }
163  }
164  }
